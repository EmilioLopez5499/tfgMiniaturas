const API_URL = 'http://localhost:3000'; function verificarAcceso(): void { const usuario = localStorage.getItem('usuario'); if (!usuario) { window.location.href = 'login.html'; return; } const datos = JSON.parse(usuario); const nombre = document.getElementById('perfil-nombre'); const email = document.getElementById('perfil-email'); if (nombre) nombre.textContent = datos.nombre; if (email) email.textContent = datos.email; } async function cargarPedidos(): Promise { const lista = document.getElementById('pedidos-lista'); if (!lista) return; const token = localStorage.getItem('token'); if (!token) return; try { const respuesta = await fetch(`${API_URL}/api/pedidos/mios`, { headers: { Authorization: `Bearer ${token}` } }); const pedidos = await respuesta.json(); if (pedidos.length === 0) { lista.innerHTML = '

No tienes pedidos todavía.

'; return; } pedidos.forEach((pedido: any) => { const card = document.createElement('div'); card.classList.add('pedido-card'); card.innerHTML = `
Pedido #${pedido.id} ${pedido.estado.replace('_', ' ')}

Creado el ${new Date(pedido.creado_en).toLocaleDateString('es-ES')}

${pedido.notas ? `

${pedido.notas}

` : ''} ${pedido.precio_total ? `

Total: ${pedido.precio_total}€

` : ''}
`; lista.appendChild(card); cargarFigurasPedido(pedido.id); }); } catch (error) { lista.innerHTML = '

Error al cargar los pedidos.

'; console.error(error); } } async function cargarFigurasPedido(pedidoId: number): Promise { const contenedor = document.getElementById(`figuras-${pedidoId}`); if (!contenedor) return; const token = localStorage.getItem('token'); if (!token) return; try { const respuesta = await fetch(`${API_URL}/api/pedidos/${pedidoId}/figuras`, { headers: { Authorization: `Bearer ${token}` } }); const figuras = await respuesta.json(); if (figuras.length === 0) return; contenedor.innerHTML = '

Figuras:

'; figuras.forEach((figura: any) => { const item = document.createElement('div'); item.classList.add('figura-item'); item.innerHTML = ` ${figura.nombre} x${figura.cantidad} ${figura.nivel_pintado ? `${figura.nivel_pintado}` : ''} `; contenedor.appendChild(item); }); } catch (error) { console.error('Error al cargar figuras del pedido:', error); } } function iniciarCerrarSesion(): void { const btn = document.getElementById('btn-cerrar-sesion'); if (!btn) return; btn.addEventListener('click', () => { localStorage.removeItem('token'); localStorage.removeItem('usuario'); window.location.href = 'index.html'; }); } document.addEventListener('DOMContentLoaded', () => { verificarAcceso(); cargarPedidos(); iniciarCerrarSesion(); });
